# Age dataset
ages <- c(30, 57, 68, 96, 39, 40, 20, 19, 42, 12, 25, 25, 65, 35, 30, 23, 23, 35, 45, 85)
mean_age <- mean(ages)

# Speed dataset
speed <- c(78.3, 81.8, 82, 74.2, 83.4, 84.5, 82.9, 77.5, 80.9, 70.6)
sd_speed <- sd(speed)
iqr_speed <- IQR(speed)

# Output
cat("Mean Age:", mean_age, "\n")
cat("Standard Deviation of Speed:", sd_speed, "\n")
cat("Interquartile Range of Speed:", iqr_speed, "\n")
