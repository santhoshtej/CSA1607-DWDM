# Load required library
library(arules)

# Load the ARFF file
data <- read.transactions("transactions.arff", format = "basket", sep = ",", cols = c(1, 2))

# Apriori Algorithm
apriori_rules <- apriori(data, parameter = list(supp = 2/9, conf = 0.5))
cat("Rules generated by Apriori:\n")
inspect(apriori_rules)

# FP-Growth Algorithm
fp_growth_rules <- eclat(data, parameter = list(supp = 2/9))
cat("Rules generated by FP-Growth:\n")
inspect(fp_growth_rules)

# Compare Unique Rules
apriori_unique <- as(apriori_rules, "data.frame")
fp_growth_unique <- as(fp_growth_rules, "data.frame")

cat("\nUnique Rules in Apriori:\n")
print(setdiff(apriori_unique, fp_growth_unique))

cat("\nUnique Rules in FP-Growth:\n")
print(setdiff(fp_growth_unique, apriori_unique))
