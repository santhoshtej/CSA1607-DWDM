# Load required library
library(arules)

# Load transaction data
transactions <- read.transactions("transactions.csv", format = "basket", sep = ",")

# Apriori Algorithm
apriori_rules <- apriori(transactions, parameter = list(supp = 0.4, conf = 0.6))
cat("Rules Generated by Apriori:\n")
inspect(apriori_rules)

# FP-Growth Algorithm
fp_growth <- eclat(transactions, parameter = list(supp = 0.4))
cat("Frequent Itemsets by FP-Growth:\n")
inspect(fp_growth)
